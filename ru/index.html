thindf — это "тонкий" формат данных (thin data format), альтернатива JSON.

Цель/назначение thindf заключается в том, чтобы почти полностью избавиться от визуального шума формата JSON (такого как символы 
{
, 
}
, 
"
и 
,
).


Примеры:
JSONthindf
{
    "personal": {
        "host": 5060852864,
        "is_team": false,
        "subscription_type": "Basic",
        "path": "C:\\Users\\DNS\\Dropbox"
    }
}


personal
    host = 5060852864
    is_team = 0B
    subscription_type = Basic
    path = C:\Users\DNS\Dropbox



{
    "added_words":
    [
        "plugin",
        "habrahabr"
    ],
    "font_size": 11
}


added_words = [

    habrahabr

font_size = 11
